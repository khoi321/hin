#include <bits/stdc++.h>
using namespace std;
int main(){
    int n,k;
    long long T;
    int u,v;
    cin >> n >> k >> T >> u >> v;
    vector<pair<int,int>> a(n);
    for(int i=0;i<n;i++) cin >> a[i].first >> a[i].second;
    vector<long long> t(n);
    for(int i=0;i<n;i++){
        long long dx=u-a[i].first;
        long long dy=v-a[i].second;
        t[i]=dx*dx+dy*dy;
    }
    vector<long long> con(t.begin()+k,t.end());
    vector<long long> lon(t.begin(),t.begin()+k);
    sort(con.begin(),con.end());
    sort(lon.begin(),lon.end());
    int res=0;
    for(long long x: con){
        if(T>=x){T-=x;res++;}else break;
    }
    for(long long x: lon){
        if(T>=x){T-=x;res++;}else break;
    }
    cout << res;
}

long long sodep(long long n) {
    long long s=0;
    for (long long i=1;i*i<=n;i++) {
        if (n%i== 0) {
            if (i%2==1) s+=i;
            if (n/i!=i && (n/i)%2==1) s+=n/i;
        }
    }
    return s;

#include <bits/stdc++.h>
using namespace std;
int main(){
    string s,res="";
    long long k;
    cin >> s >> k;
    int i=0;
    while(i<s.size()){
        if(isdigit(s[i])){
            int j=i;
            while(j<s.size() && isdigit(s[j])) j++;
            int val=0;
            for(int t=i;t<j;t++) val=val*10+s[t]-'0';
            res+=char(val);
            i=j;
        } else {
            char c=s[i];
            if(isupper(c)){
                long long shift=k%26;
                char orig=(c-'A'-shift+26)%26+'A';
                res+=orig;
            } else {
                long long shift=(k-1)%26;
                char orig=(c-'a'-shift+26)%26+'a';
                res+=orig;
            }
            i++;
        }
    }
    cout << res;
}

#include <bits/stdc++.h>
using namespace std;

int main() {
    int n;
    cin >> n;
    vector<int> h(n);
    for (int i = 0; i < n; i++) {
        cin >> h[i];
    }
    int maxLength = 0;
    
    for (int i = 1; i < n - 1; i++) {
        if (h[i] > h[i-1] && h[i] > h[i+1]) {
            int left = i - 1;
            int right = i + 1;
            
            while (left > 0 && h[left-1] < h[left]) {
                left--;
            }
            while (right < n - 1 && h[right] > h[right+1]) {
                right++;
            }
            int length = right - left + 1;
            if (length >= 3) {
                maxLength = max(maxLength, length);
            }
        }
    }
    
    cout << maxLength << endl;
    return 0;
}
